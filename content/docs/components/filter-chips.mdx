## Features

- **Shared layout transitions** — Smooth background animation between chips
- **Spring physics** — Natural, bouncy selection feedback
- **Multiple variants** — default, outline, ghost, pill styles
- **Type-safe** — Full TypeScript generics support

## Interactions

| Trigger | Action |
|---------|--------|
| Click Chip | Active indicator animates to clicked chip |
| Hover | Background lightens subtly |
| Focus | Keyboard accessible with Tab navigation |
| Enter/Space | Activates focused chip |

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `filters` | `Array<{value, label}>` | - | Array of filter options |
| `activeFilter` | `T` | - | Currently active filter value |
| `onFilterChange` | `(value: T) => void` | - | Callback when filter changes |
| `variant` | `"default" \| "outline" \| "ghost" \| "pill"` | `"pill"` | Visual style |
| `size` | `"sm" \| "md" \| "lg"` | `"md"` | Chip size |

## Usage

```tsx
import { FilterChips } from "@/components/ui/filter-chips";

const [active, setActive] = useState("all");

const filters = [
  { value: "all", label: "All" },
  { value: "design", label: "Design" },
  { value: "dev", label: "Development" },
];

<FilterChips
  filters={filters}
  activeFilter={active}
  onFilterChange={setActive}
  variant="pill"
/>
```
